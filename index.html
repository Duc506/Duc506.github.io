import axios from 'axios'; const API_URL = 'http://example.com/api'; //địa chỉ của API trên trang web const LOGIN_URL = `${API_URL}/login`; //Địa chỉ đăng nhập axios.post(LOGIN_URL, { //Request đăng nhập bằng phương thức POST email: 'your-email@example.com', //nhập email để đăng nhập password: 'your-password' //nhập password để đăng nhập }).then(response => { const token = response.data.token; //lấy token trả về từ API axios.defaults.headers.common['Authorization'] = `Bearer ${token}`; //lưu token vào Headers cho các request tiếp theo }).catch(error => { console.error(error); });
